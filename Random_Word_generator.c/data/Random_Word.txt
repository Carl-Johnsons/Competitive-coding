    printf("\t\t\t\t\t\t|1.Random letters only        |\n");
    printf("\t\t\t\t\t\t|2.Random words only          |\n");
    printf("\t\t\t\t\t\t|0.Exit the program           |\n");
    printf("\t\t\t\t\t\t==============================\n");
    char option;
    int ascii;
    do
    {
        option = getch();
        ascii = option;
    } while (option != '1' && option != '2' && option != '0');
    system("cls");
    unsigned long long n;
    int ran;
    FILE *sc;
    sc = fopen("src.doc", "r");
    char string001[20] = {0};

    }
    int len = 1211;
    srand(time(NULL));
    FILE *fp = NULL;
    fp = fopen("RandomWord.txt", "w");

    if (option == '1')
    {
        printf("\t\t\t\t\t\tRANDOM LETTERS!!!\n\n");

            fprintf(fp, "%c", ran);

        }
    }

int random123(int min, int max)
{
    return rand() % (max - min + 1) + min;
}
